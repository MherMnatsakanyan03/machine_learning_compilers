For compiling assembly in wsl2:

sudo apt install g++-aarch64-linux-gnu qemu-user-static qemu-user

aarch64-linux-gnu-g++ main.cpp assemb.s -o main.exe -static

options:
- sve-default-vector-length=64 (in bytes)

qemu-aarch64 ./main.exe
